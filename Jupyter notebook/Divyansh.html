def mean(array):
    return sum(array)/len(array)

def std(array):
    avg = mean(array)
    variance = 0
    for value in array:
        variance += (avg - value)**2
    variance = variance/len(array)
    return round(variance**0.5,2)

def median(array):
    array.sort()
    if len(array)%2==0:
        index = int(len(array)/2)
        return (array[index] + array[index-1])/2
    else:
        index = int((len(array)-1)/2)
        return array[index]

3. Difference between mutable and immutable datatypes.¶

Mutable objects can have their contents modified after they are created. You can change, add, or remove elements without creating a new object. When global mutable objects are passed into local spaces (e.g. passed into functions), changes in the local environment are reflected globally. e.g. - Lists, Dictionary, Sets etc.

Immutable objects cannot be modified after they are created. Any operation that seems to modify them actually creates a new object. Changes in local environment are not reflected globally. e.g. - String, Tuple, Integers, Floats etc.

def fibonacciList(n):
    array = []
    pen = 1
    ult = 1
    for i in range(n):
        if i==0:
            array.append(pen)
            continue
        array.append(ult)
        pen,ult = ult,pen+ult
    return array

    sortedWord = sorted(word)
    return [w for w in words if sorted(w) == sortedWord]

import matplotlib.pyplot as plt

def mean(array):
    return sum(array)/len(array)

def std(array):
    avg = mean(array)
    variance = 0
    for value in array:
        variance += (avg - value)**2
    variance = variance/len(array)
    return round(variance**0.5,2)

def median(array):
    array.sort()
    if len(array)%2==0:
        index = int(len(array)/2)
        return (array[index] + array[index-1])/2
    else:
        index = int((len(array)-1)/2)
        return array[index]

def mode(array):
    hashmap = {}
    for i in array:
        if i in hashmap:
            hashmap[i] += 1
        else:
            hashmap[i] = 1
    return max(hashmap , key = hashmap.get)

import matplotlib.pyplot as plt
def normalise(array):
    handle = []
    minimum = min(array)
    maximum = max(array)
    for i in array:
        n = (i - minimum)/(maximum - minimum)
        handle.append(n)
    return handle

array = [12,13,35,3,67,86,34,56,90,99,12,100,7,23,54]
normalised = normalise(array)
print(normalised)
